Core Types

BackgammonColor

Represents the two player colors in backgammon.

type BackgammonColor = 'black' | 'white'
Usage Example
const playerColor: BackgammonColor = 'black'
const opponentColor: BackgammonColor = 'white'

BackgammonMoveDirection

Defines the direction of movement for each player around the board.

type BackgammonMoveDirection = 'clockwise' | 'counterclockwise'
Usage Example
// Black player moves clockwise (1‚Üí24)
const blackDirection: BackgammonMoveDirection = 'clockwise'

// White player moves counterclockwise (24‚Üí1)  
const whiteDirection: BackgammonMoveDirection = 'counterclockwise'

BackgammonDieValue

Represents the possible values on a backgammon die.

type BackgammonDieValue = 1 | 2 | 3 | 4 | 5 | 6
Usage Example
const dieValue: BackgammonDieValue = 6
const roll: [BackgammonDieValue, BackgammonDieValue] = [3, 5]

interface BackgammonBoard { id: string BackgammonPoints: BackgammonPoints // Array of 24 points bar: { clockwise: BackgammonBar counterclockwise: BackgammonBar } off: { clockwise: BackgammonOff counterclockwise: BackgammonOff } }

BackgammonPlayerStateKind

All possible player states in the game flow.

type BackgammonPlayerStateKind = | 'inactive' | 'rolling-for-start' | 'rolled-for-start' | 'rolling' | 'rolled' | 'moving' | 'moved' | 'winner' | 'doubled'
Type Guards
function isPlayerRolled(player: BackgammonPlayer): player is BackgammonPlayerRolled {
  return player.stateKind === 'rolled'
}

All possible game states representing the complete game flow.

BackgammonGameMoving

Game state when a player is actively making moves.

interface BackgammonGameMoving extends BackgammonGame { stateKind: 'moving' activeColor: BackgammonColor activePlayer: BackgammonPlayerMoving inactivePlayer: BackgammonPlayerInactive activePlay: BackgammonPlayMoving }
Type-Safe Game Processing
function processGame(game: BackgammonGame): void {
  if (game.stateKind === 'moving') {
    // TypeScript knows this is BackgammonGameMoving
    const activePlay = game.activePlay
    const moves = activePlay.moves
    const availableDice = activePlay.availableDice
  }
}

Complete move definition with all necessary information for execution and validation.

interface BackgammonMove { id: string kind: BackgammonMoveKind origin: BackgammonMoveOrigin destination: BackgammonMoveDestination dieValue: BackgammonDieValue checker: BackgammonChecker hitChecker?: BackgammonChecker // If opponent checker was hit }

BackgammonMoveKind

The three types of moves in backgammon.

type BackgammonMoveKind = 'point-to-point' | 'bear-off' | 'reenter'

Specific Move Types

Concrete move implementations for each move kind.

interface BackgammonMovePointToPoint extends BackgammonMove { kind: 'point-to-point' origin: BackgammonPoint destination: BackgammonPoint } interface BackgammonMoveBearOff extends BackgammonMove { kind: 'bear-off' origin: BackgammonPoint // Must be in home board destination: BackgammonOff } interface BackgammonMoveReenter extends BackgammonMove { kind: 'reenter' origin: BackgammonBar destination: BackgammonPoint // Must be in opponent's home board }
Move Type Checking
function handleMove(move: BackgammonMove): void {
  switch (move.kind) {
    case 'point-to-point':
      // TypeScript knows this is BackgammonMovePointToPoint
      const fromPoint = move.origin.position
      const toPoint = move.destination.position
      break
    case 'bear-off':
      // TypeScript knows this is BackgammonMoveBearOff
      const homePoint = move.origin.position
      break
    case 'reenter':
      // TypeScript knows this is BackgammonMoveReenter
      const entryPoint = move.destination.position
      break
  }
}

BackgammonDiceRolled

Dice state after rolling with current roll values.

interface BackgammonDiceRolled { id: string stateKind: 'rolled' color: BackgammonColor currentRoll: BackgammonRoll total: number }

BackgammonRoll

Tuple representing the values of both dice.

type BackgammonRoll = [BackgammonDieValue, BackgammonDieValue]
Doubles Handling
function isDouble(roll: BackgammonRoll): boolean {
  return roll[0] === roll[1]
}

function getAvailableMoves(roll: BackgammonRoll): BackgammonDieValue[] {
  if (isDouble(roll)) {
    // Doubles give 4 moves of the same value
    return [roll[0], roll[0], roll[0], roll[0]]
  }
  return [...roll]
}

Type-Safe Game Creation

import type { 
  BackgammonGame, 
  BackgammonPlayer,
  BackgammonBoard
} from '@nodots-llc/backgammon-types'

// All types are enforced at compile time
const createGame = (
  player1: BackgammonPlayer, 
  player2: BackgammonPlayer,
  board?: BackgammonBoard
): BackgammonGame => {
  return {
    id: generateId(),
    stateKind: 'rolling-for-start',
    players: [player1, player2],
    board: board ?? Board.initialize(),
    cube: Cube.initialize()
  }
}

Discriminated Union Pattern

function handlePlayerAction(player: BackgammonPlayer): void {
  switch (player.stateKind) {
    case 'rolling':
      // TypeScript knows: BackgammonPlayerRolling
      console.log('Player is ready to roll')
      break
    case 'rolled':
      // TypeScript knows: BackgammonPlayerRolled
      const roll = player.dice.currentRoll
      console.log(`Rolled: ${roll[0]}, ${roll[1]}`)
      break
    case 'moving':
      // TypeScript knows: BackgammonPlayerMoving
      console.log('Player is making moves')
      break
    case 'winner':
      // TypeScript knows: BackgammonPlayerWinner
      console.log(`${player.color} wins!`)
      break
    default:
      // Exhaustiveness check ensures all cases handled
      const _exhaustive: never = player.stateKind
  }
}

Type Guards & Validation

// Runtime type checking with type guards
function isGameMoving(game: BackgammonGame): game is BackgammonGameMoving {
  return game.stateKind === 'moving'
}

function isBearOffMove(move: BackgammonMove): move is BackgammonMoveBearOff {
  return move.kind === 'bear-off'
}

// Safe move processing
function processMove(game: BackgammonGame, move: BackgammonMove): void {
  if (isGameMoving(game) && isBearOffMove(move)) {
    // Both types are narrowed safely
    const activePlay = game.activePlay
    const homePoint = move.origin.position
    console.log(`Bearing off from point ${homePoint.clockwise}`)
  }
}
