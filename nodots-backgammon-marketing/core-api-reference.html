Board Class

Board.initialize()

Creates a new backgammon board with standard starting positions.

static initialize(): BackgammonBoard
Example
import { Board } from '@nodots-llc/backgammon-core'

const board = Board.initialize()
console.log(board.BackgammonPoints.length) // 24

Board.moveChecker()

Executes a checker move from one position to another, handling hits and validation.

static moveChecker( board: BackgammonBoard, move: BackgammonMove ): BackgammonBoard
board
BackgammonBoard
Current board state
move
BackgammonMove
Move to execute (PointToPoint, BearOff, or Reenter)
Example
const move = {
  id: 'move-123',
  kind: 'point-to-point',
  origin: board.BackgammonPoints[12],
  destination: board.BackgammonPoints[18],
  dieValue: 6,
  checker: someChecker
}

const newBoard = Board.moveChecker(board, move)

Board.getPossibleMoves()

Calculates all legal moves for a player given a specific die value.

static getPossibleMoves( board: BackgammonBoard, player: BackgammonPlayer, dieValue: BackgammonDieValue ): BackgammonMoveSkeleton[]
board
BackgammonBoard
Current board state
player
BackgammonPlayer
Player to calculate moves for
dieValue
BackgammonDieValue
Die value to use (1-6)
Example
const possibleMoves = Board.getPossibleMoves(board, player, 6)
console.log(`${possibleMoves.length} moves available for die value 6`)

// Filter by move type
const bearOffMoves = possibleMoves.filter(move => move.kind === 'bear-off')

Board.getPipCounts()

Calculates pip counts for both players on the board.

static getPipCounts( board: BackgammonBoard ): { clockwise: number, counterclockwise: number }
Example
const pipCounts = Board.getPipCounts(board)
console.log(`Black pip count: ${pipCounts.clockwise}`)
console.log(`White pip count: ${pipCounts.counterclockwise}`)

Player Class

Player.initialize()

Creates a new player with specified color and movement direction.

static initialize( color: BackgammonColor, direction: BackgammonMoveDirection ): BackgammonPlayer
color
BackgammonColor
'black' or 'white'
direction
BackgammonMoveDirection
'clockwise' or 'counterclockwise'
Example
const player1 = Player.initialize('black', 'clockwise')
const player2 = Player.initialize('white', 'counterclockwise')

console.log(player1.stateKind) // 'inactive'

Player.roll()

Rolls dice for a player in the rolling state.

static roll(player: BackgammonPlayerRolling): BackgammonPlayerRolled
Example
// Player must be in 'rolling' state
if (player.stateKind === 'rolling') {
  const rolledPlayer = Player.roll(player)
  const [die1, die2] = rolledPlayer.dice.currentRoll
  console.log(`Rolled: ${die1}, ${die2}`)
}

Player.getHomeBoard()

Returns the home board points for the player based on their movement direction.

static getHomeBoard( player: BackgammonPlayer, board: BackgammonBoard ): BackgammonPoint[]
Example
const homeBoard = Player.getHomeBoard(player, board)
console.log(`Home board has ${homeBoard.length} points`) // 6

// Check if all checkers are in home board
const checkersInHome = homeBoard.every(point => 
  point.checkers.every(checker => checker.color === player.color)
)

Game Class

Game.initialize()

Creates a new backgammon game with two players.

static initialize(players: BackgammonPlayers): BackgammonGame
Example
const player1 = Player.initialize('black', 'clockwise')
const player2 = Player.initialize('white', 'counterclockwise')
const game = Game.initialize([player1, player2])

console.log(game.stateKind) // 'rolling-for-start'

Game.rollForStart()

Determines which player goes first by rolling dice.

static rollForStart(game: BackgammonGameRollingForStart): BackgammonGame
Example
if (game.stateKind === 'rolling-for-start') {
  const startedGame = Game.rollForStart(game)
  // Game state changes to 'rolled-for-start' or 'rolling' depending on result
}

Game.roll()

Rolls dice for the active player.

static roll(game: BackgammonGameRolling): BackgammonGameRolled
Example
if (game.stateKind === 'rolling') {
  const rolledGame = Game.roll(game)
  const activePlayer = rolledGame.activePlayer
  const [die1, die2] = activePlayer.dice.currentRoll
}

Game.move()

Executes moves within the game context.

static move( game: BackgammonGameMoving, moves: BackgammonMove[] ): BackgammonGame
Example
if (game.stateKind === 'moving') {
  const moves = [/* calculated moves */]
  const updatedGame = Game.move(game, moves)
}

Move System

Move Types

Three types of moves are supported in backgammon.

type BackgammonMoveKind = 'point-to-point' | 'bear-off' | 'reenter'
Point-to-Point Move
const pointToPointMove: BackgammonMovePointToPoint = {
  id: 'move-1',
  kind: 'point-to-point',
  origin: board.BackgammonPoints[12],
  destination: board.BackgammonPoints[18],
  dieValue: 6,
  checker: someChecker
}
Bear-Off Move
const bearOffMove: BackgammonMoveBearOff = {
  id: 'move-2',
  kind: 'bear-off',
  origin: board.BackgammonPoints[18], // Must be in home board
  destination: board.off.clockwise,
  dieValue: 6,
  checker: homeChecker
}
Reenter Move
const reenterMove: BackgammonMoveReenter = {
  id: 'move-3',
  kind: 'reenter',
  origin: board.bar.clockwise,
  destination: board.BackgammonPoints[2], // Must be open
  dieValue: 3,
  checker: hitChecker
}

Move Validation

All moves are validated according to backgammon rules.

Validation Rules
// Point-to-Point: Destination must be open or contain friendly checkers
// Bear-Off: Checker must be in home board, no checkers behind
// Reenter: Must reenter from bar before other moves
// Hit: Opponent checker is sent to bar

const moves = Board.getPossibleMoves(board, player, dieValue)
// Only returns legal moves that pass validation

Complete Examples

Basic Game Flow

import { Board, Player, Game } from '@nodots-llc/backgammon-core'

// 1. Initialize game
const board = Board.initialize()
const player1 = Player.initialize('black', 'clockwise')
const player2 = Player.initialize('white', 'counterclockwise')
const game = Game.initialize([player1, player2])

// 2. Roll for start
let currentGame = Game.rollForStart(game)

// 3. Game loop
while (currentGame.stateKind !== 'completed') {
  if (currentGame.stateKind === 'rolling') {
    currentGame = Game.roll(currentGame)
  } else if (currentGame.stateKind === 'moving') {
    // Calculate and execute moves
    const possibleMoves = Board.getPossibleMoves(
      currentGame.board,
      currentGame.activePlayer,
      currentGame.activePlayer.dice.currentRoll[0]
    )
    
    if (possibleMoves.length > 0) {
      currentGame = Game.move(currentGame, [possibleMoves[0]])
    }
  }
}

Error Handling

try {
  const moves = Board.getPossibleMoves(board, player, 6)
  
  if (moves.length === 0) {
    console.log('No legal moves available')
    return
  }
  
  const newBoard = Board.moveChecker(board, moves[0])
  
} catch (error) {
  if (error instanceof BackgammonValidationError) {
    console.log('Invalid move:', error.message)
  } else {
    console.log('Unexpected error:', error)
  }
}
